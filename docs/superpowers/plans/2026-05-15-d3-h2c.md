# D.3 — H2C (Cleartext HTTP/2) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add `conn.PlaintextDialer` for H2C prior-knowledge connections (RFC 7540 §3.4) and `ClientOptions.DefaultScheme` so callers targeting H2C backends set `:scheme: http` once at client construction.

**Architecture:** `PlaintextDialer` dials raw TCP and returns the `net.Conn` directly — the existing `NewClientConn` already sends the HTTP/2 client preface and runs the SETTINGS handshake over any `net.Conn`, so the conn layer needs no changes. `ClientOptions.DefaultScheme` is stored on `Client` as `defaultScheme string` and threaded into `buildHeaders` as a third parameter, replacing the hard-coded `"https"` fallback.

**Tech Stack:** Go 1.24, `golang.org/x/net/http2` + `golang.org/x/net/http2/h2c` (already in go.mod) for h2c test servers.

## Background: what already exists (do NOT re-implement)

```go
package conn

import (

)

// startH2CServer starts an H2C (cleartext HTTP/2) server on a random
// port and returns the "host:port" address. The server is shut down
// when the test ends.
func startH2CServer(t *testing.T, handler http.Handler) string {
	t.Helper()
	ln, err := net.Listen("tcp", "127.0.0.1:0")
	if err != nil {
		t.Fatalf("listen: %v", err)
	}
	srv := &http.Server{Handler: h2c.NewHandler(handler, &xhttp2.Server{})}
	go func() { _ = srv.Serve(ln) }()
	t.Cleanup(func() { _ = srv.Close() })
	return ln.Addr().String()
}

func TestPlaintextDialer_H2C(t *testing.T) {
	addr := startH2CServer(t, http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
		w.WriteHeader(200)
	}))

	ctx, cancel := context.WithTimeout(context.Background(), 5*time.Second)
	defer cancel()

	c, err := Dial(ctx, addr, ConnOptions{Dialer: &PlaintextDialer{}})
	if err != nil {
		t.Fatalf("Dial: %v", err)
	}
	defer c.Close()

	s, err := c.NewStream(ctx)
	if err != nil {
		t.Fatalf("NewStream: %v", err)
	}
	if err := s.SendHeaders(ctx, []hpack.HeaderField{
		{Name: []byte(":method"), Value: []byte("GET")},
		{Name: []byte(":scheme"), Value: []byte("http")},
		{Name: []byte(":authority"), Value: []byte(addr)},
		{Name: []byte(":path"), Value: []byte("/")},
	}, true); err != nil {
		t.Fatalf("SendHeaders: %v", err)
	}

	var status string
	for {
		ev, err := s.Recv(ctx)
		if err != nil {
			t.Fatalf("Recv: %v", err)
		}
		if ev.Type == EventHeaders {
			for _, f := range ev.Headers {
				if string(f.Name) == ":status" {
					status = string(f.Value)
				}
			}
		}
		if ev.EndStream {
			break
		}
	}
	if status != "200" {
		t.Fatalf("status = %q, want %q", status, "200")
	}
}
```

```go
// PlaintextDialer dials addr over TCP for H2C prior-knowledge connections
// (RFC 7540 §3.4). No TLS handshake or ALPN negotiation is performed.
// NewClientConn sends the HTTP/2 connection preface automatically.
type PlaintextDialer struct{}

// Dial dials addr over TCP and returns the raw connection.
func (d *PlaintextDialer) Dial(ctx context.Context, addr string) (net.Conn, error) {
	var nd net.Dialer
	return nd.DialContext(ctx, "tcp", addr)
}
```

```go
	t.Helper()

	if err != nil {
		t.Fatalf("listen: %v", err)
	}
	srv := &http.Server{Handler: h2c.NewHandler(h, &xhttp2.Server{})}
	go func() { _ = srv.Serve(ln) }()
	t.Cleanup(func() { _ = srv.Close() })

func TestIntegration_Client_H2C_Do(t *testing.T) {
	addr := newH2CServer(t, http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
		w.WriteHeader(200)
		_, _ = io.WriteString(w, "h2c ok")
	}))

	ctx, cancel := context.WithTimeout(context.Background(), 5*time.Second)
	defer cancel()

	c, err := client.NewClient(client.ClientOptions{
		Addr: addr,
		ConnOpts: conn.ConnOptions{
			Dialer: &conn.PlaintextDialer{},
		},
		DefaultScheme: "http",
	})
	if err != nil {
		t.Fatalf("NewClient: %v", err)
	}
	t.Cleanup(func() { _ = c.Close() })

	var res client.Response
	err = c.Do(ctx, &client.Request{Method: "GET", Path: "/"}, &res)
	if err != nil {
		t.Fatalf("Do: %v", err)
	}
	if res.Status != 200 {
		t.Fatalf("status = %d, want 200", res.Status)
	}
```
